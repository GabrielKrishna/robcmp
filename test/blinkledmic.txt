/* blink a set of LEDSs on arduino, based on
 * the noise in a microphone
 * thiago 2018/1 compiler course
 */

noise = 0;
i = 0;
while (i < 10) {
  noise = noise + in3;
  out9 = 255;
  delay(100);
  out9 = 0;
  delay(100);
  i = i + 1;
}
noise = noise / 10;

while (1 > 0) {
    a = in3 - noise;
    
    if (a < 0)
      a = 0 - a;
    
    if (a > 20) {
      out9 = a;
      if (a > 33) {
        out5 = a;
	  }
      if (a > 66) {
        out3 = a;
	  }
    } else {
      out9 = 0;
      out5 = 0;
      out3 = 0;
    }
}
